CREATE SEQUENCE singer_seq START WITH 1 INCREMENT BY 1;
CREATE SEQUENCE song_seq START WITH 1 INCREMENT BY 1;

CREATE TABLE Singer (
  id INT DEFAULT singer_seq.NEXTVAL PRIMARY KEY,
  name NVARCHAR2(10),
  hit_song_id INT
);

CREATE TABLE Song (
  id INT DEFAULT song_seq.NEXTVAL PRIMARY KEY,
  title NVARCHAR2(20) NOT NULL,
  price INT
);
