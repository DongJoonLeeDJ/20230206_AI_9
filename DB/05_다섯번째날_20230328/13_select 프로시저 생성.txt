CREATE OR REPLACE PROCEDURE PERSON_SELECT(
MYNAME IN VARCHAR2,
O_CURSOR OUT SYS_REFCURSOR)
    IS BEGIN OPEN O_CURSOR 
    FOR 
    SELECT NUM,NAME,HP FROM SIMPLE_TEST_TABLE WHERE 
    NAME = MYNAME;
    EXCEPTION WHEN OTHERS THEN DBMS_OUTPUT.PUT_LINE(
    'SQL ERROR' || SQLERRM);
END PERSON_SELECT; 
/
