-- 수정 : update 문
-- UPDATE, DELETE 주의사항 : 반드시 WHERE을 써줘야 한다.
-- 안 쓰면 일괄 변경 OR 일괄 삭제가 됩니다.
-- WHERE문을 잘 써야 된다.
UPDATE mysoldier SET GENDER = 'M';
--10개 행 이(가) 업데이트되었습니다.
-- 모든 군인의 성별을 남성으로 일괄 변경한 것

-- 이영희가 여자도 있고 남자도 있는 경우에 이 코드는 부적절
UPDATE mysoldier SET GENDER='W' WHERE NAME='이영희';



UPDATE mysoldier SET GENDER='W' WHERE NAME='이영희' OR NAME = '박성호';
-- IN
-- 여러 명 성별 변환 가능
UPDATE mysoldier SET GENDER='W' WHERE NAME IN('강영민','이승민','박준영');
