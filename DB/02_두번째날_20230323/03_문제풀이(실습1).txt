--TRIM 같은 함수를 통해서 공백을 제거하고 다시 업데이트를 해도 됨

-- 공백이 싹 지워짐
-- 앞뒤의 공백을 지우는 것
UPDATE mysoldier SET NAME=TRIM(NAME);
SELECT * FROM mysoldier;

-- DROP TABLE
-- 테이블을 삭제하는 키워드가 3개

-- DELETE 
-- WHERE문을 잘 써서 다 지워지는 걸 방지해야 한다.
-- 다행스럽게도 ROLLBACK등으로 복구가 가능
COMMIT;
DELETE FROM MYSOLDIER;
SELECT * FROM mysoldier;
ROLLBACK;

-- TRUNCATE
-- DELETE가 휴지통에 넣는 거라면 TRUNCATE는 SHIFT+DELETE(완전 삭제)
COMMIT;
TRUNCATE TABLE MYSOLDIER;
-- Table MYSOLDIER이(가) 잘렸습니다.
ROLLBACK;


-- DROP
-- 테이블 자체를 완전히 지워버림
